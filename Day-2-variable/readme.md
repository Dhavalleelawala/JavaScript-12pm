
# Arithmetic operators
 + (addition)
 - (subtraction)
 * (multiplication)
 / (division)
 % (modulus)
 ** (exponentiation)

# Assignment operators
 = (assignment)
 += (addition assignment)
 -= (subtraction assignment)
 *= (multiplication assignment)
 /= (division assignment)
 %= (modulus assignment)
 **= (exponentiation assignment)

# Comparison operators
 == (equal to)
 != (not equal to)
 === (strictly equal to)
 !== (strictly not equal to)
 >
 <
 >=
 <=

# Logical operators
 && (logical and)
 || (logical or)
 ! (logical not)
